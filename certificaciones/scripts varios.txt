/*
====================================================================================
 PARTE 1: SCRIPT DE CONSOLIDACIÓN (POSTGRESQL 10.23)
====================================================================================
NOTA: Las tablas y columnas ya existen en producción. Este script se enfoca 
estrictamente en asegurar la integridad referencial (Foreign Keys) y los permisos.
*/

DO $$
BEGIN
    -- 1. Asegurar FK en usuario_documentos
    IF NOT EXISTS (SELECT 1 FROM pg_constraint WHERE conname = 'fk_usuario_documentos_curso') THEN
        ALTER TABLE cursos.usuario_documentos 
        ADD CONSTRAINT fk_usuario_documentos_curso FOREIGN KEY (id_curso) REFERENCES cursos.cursos (id_curso) ON DELETE SET NULL;
    END IF;

    -- 2. Asegurar FK en buzon_sugerencias
    IF NOT EXISTS (SELECT 1 FROM pg_constraint WHERE conname = 'buzon_sugerencias_id_usuario_fkey') THEN
        ALTER TABLE cursos.buzon_sugerencias 
        ADD CONSTRAINT buzon_sugerencias_id_usuario_fkey FOREIGN KEY (id_usuario) REFERENCES cursos.usuarios (id) ON DELETE SET NULL;
    END IF;

    -- 3. Asegurar FK en comprobantes_pago
    IF NOT EXISTS (SELECT 1 FROM pg_constraint WHERE conname = 'fk_comprobantes_pago_materia') THEN
        ALTER TABLE cursos.comprobantes_pago 
        ADD CONSTRAINT fk_comprobantes_pago_materia FOREIGN KEY (id_materia_bimestre) REFERENCES cursos.materias_bimestre (id_materia_bimestre) ON DELETE CASCADE;
    END IF;

    -- Nota: Si faltan más FK de las tablas base (como materias_bimestre -> cursos), 
    -- deben agregarse a este bloque usando la misma lógica.
END $$;

-- 4. Asegurar todos los privilegios de forma masiva
GRANT ALL PRIVILEGES ON TABLE 
    cursos.usuario_documentos, 
    cursos.materias_bimestre, 
    cursos.actividades_config, 
    cursos.notas_participante, 
    cursos.actas_cierre, 
    cursos.cuentas_bancarias, 
    cursos.comprobantes_pago, 
    cursos.buzon_sugerencias, 
    cursos.landing_carrusel 
TO uptaivir_certificacion2;

GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA cursos TO uptaivir_certificacion2;

/*
====================================================================================
 PARTE 2: DDL CONSOLIDADO DE LA BASE DE DATOS (REFERENCIA COMPLETA)
====================================================================================
NOTA: Este script crea TODA la base de datos de cero, con todas sus tablas, 
relaciones, triggers y funciones ya consolidadas sin redundancias. Ideal para 
montar un entorno local nuevo o por si se necesita recrear la BD entera en el futuro.
*/

-- Configuración inicial
SET statement_timeout = 0;
SET lock_timeout = 0;
SET idle_in_transaction_session_timeout = 0;
SET client_encoding = 'UTF8';
SET standard_conforming_strings = on;
SELECT pg_catalog.set_config('search_path', '', false);
SET check_function_bodies = false;
SET xmloption = content;
SET client_min_messages = warning;
SET row_security = off;

-- Crear esquema
CREATE SCHEMA IF NOT EXISTS cursos AUTHORIZATION CURRENT_USER;
SET search_path TO cursos;

-- Crear extensión para UUID
CREATE EXTENSION IF NOT EXISTS "uuid-ossp" WITH SCHEMA public;

-- ==================== TABLAS INDEPENDIENTES ====================

CREATE TABLE IF NOT EXISTS config_sistema (
    clave_config VARCHAR(100) PRIMARY KEY,
    valor_config VARCHAR(255) NOT NULL,
    descripcion_config TEXT NULL,
    fecha_modificacion TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS roles (
    id_rol SERIAL PRIMARY KEY,
    nombre_rol VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS posiciones_firma (
    id_posicion SERIAL PRIMARY KEY,
    codigo_posicion VARCHAR(25) UNIQUE NOT NULL,
    descripcion_posicion VARCHAR(100) NOT NULL,
    pagina SMALLINT NOT NULL
);

CREATE TABLE IF NOT EXISTS cursos (
    id_curso SERIAL PRIMARY KEY,
    promotor VARCHAR(255),
    nombre_curso VARCHAR(255),
    descripcion TEXT,
    tiempo_asignado INTEGER,
    inicio_mes DATE,
    tipo_curso VARCHAR(255),
    autorizacion VARCHAR(255),
    limite_inscripciones INTEGER,
    estado BOOLEAN,
    dias_clase TEXT[],
    horario_inicio TIME WITHOUT TIME ZONE,
    horario_fin TIME WITHOUT TIME ZONE,
    nivel_curso VARCHAR(255),
    costo NUMERIC(10,2),
    conocimientos_previos TEXT,
    requerimientos_implemento TEXT,
    desempeno_al_concluir TEXT,
    horas_cronologicas INTEGER,
    firma_digital BOOLEAN DEFAULT FALSE,
    fecha_finalizacion TIMESTAMP WITHOUT TIME ZONE
);

CREATE TABLE IF NOT EXISTS cuentas_bancarias (
    id_cuenta SERIAL PRIMARY KEY,
    banco VARCHAR(100) NOT NULL,
    titular VARCHAR(100) NOT NULL,
    cedula_rif VARCHAR(50) NOT NULL,
    telefono VARCHAR(50), 
    correo VARCHAR(100), 
    tipo_cuenta VARCHAR(50), 
    numero_cuenta VARCHAR(50),
    activo BOOLEAN NOT NULL DEFAULT TRUE
);

CREATE TABLE IF NOT EXISTS auditoria (
    id_auditoria SERIAL PRIMARY KEY,
    usuario VARCHAR(255),
    accion VARCHAR(20) NOT NULL,
    tabla_afectada VARCHAR(50) NOT NULL,
    fecha TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    dato_previo TEXT,
    dato_modificado TEXT
);

-- ==================== TABLAS DEPENDIENTES ====================

CREATE TABLE IF NOT EXISTS usuarios (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100),
    apellido VARCHAR(100),
    correo VARCHAR(100) UNIQUE,
    password TEXT,
    cedula VARCHAR(100) UNIQUE,
    id_rol INTEGER REFERENCES roles(id_rol),
    token VARCHAR(255),
    confirmado BOOLEAN,
    firma_digital VARCHAR(255),
    titulo VARCHAR(50),
    cargo VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS cargos (
    id_cargo SERIAL PRIMARY KEY,
    nombre_cargo VARCHAR(100) NOT NULL,
    nombre VARCHAR(100),
    apellido VARCHAR(100),
    titulo VARCHAR(50),
    firma_digital VARCHAR(255),
    id_usuario INTEGER REFERENCES usuarios(id),
    activo BOOLEAN NOT NULL DEFAULT TRUE
);

CREATE TABLE IF NOT EXISTS buzon_sugerencias (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100),
    apellido VARCHAR(100),
    correo VARCHAR(100),
    cedula VARCHAR(100),
    sugerencia TEXT,
    fecha_envio TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    id_usuario INTEGER REFERENCES usuarios(id)
);

CREATE TABLE IF NOT EXISTS modulos (
    id_modulo SERIAL PRIMARY KEY,
    id_curso INTEGER REFERENCES cursos(id_curso) ON DELETE CASCADE,
    nombre_modulo VARCHAR(255),
    contenido TEXT,
    numero INTEGER,
    actividad VARCHAR(255),
    instrumento VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS usuario_documentos (
    documento_id SERIAL PRIMARY KEY,
    usuario_id INTEGER NOT NULL REFERENCES usuarios(id),
    documento_path VARCHAR(255) NOT NULL,
    documento_type VARCHAR(50) NOT NULL,
    id_curso INTEGER REFERENCES cursos(id_curso) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS certificaciones (
    id_certificacion SERIAL PRIMARY KEY,
    id_usuario INTEGER REFERENCES usuarios(id),
    curso_id INTEGER REFERENCES cursos(id_curso),
    valor_unico VARCHAR UNIQUE,
    completado BOOLEAN DEFAULT FALSE,
    nota INTEGER,
    fecha_inscripcion TIMESTAMP WITHOUT TIME ZONE,
    pago BOOLEAN,
    tomo INTEGER,
    folio INTEGER
);

CREATE TABLE IF NOT EXISTS cursos_config_firmas (
    id_config SERIAL PRIMARY KEY,
    id_curso INTEGER NOT NULL REFERENCES cursos(id_curso) ON DELETE CASCADE,
    id_posicion INTEGER NOT NULL REFERENCES posiciones_firma(id_posicion) ON DELETE RESTRICT,
    id_cargo_firmante INTEGER REFERENCES cargos(id_cargo) ON DELETE SET NULL,
    usar_promotor_curso BOOLEAN NOT NULL DEFAULT FALSE,
    CONSTRAINT chk_firmante_exclusivo CHECK (
        (id_cargo_firmante IS NOT NULL AND usar_promotor_curso = FALSE) OR 
        (id_cargo_firmante IS NULL AND usar_promotor_curso = TRUE) OR      
        (id_cargo_firmante IS NULL AND usar_promotor_curso = FALSE)
    ),
    CONSTRAINT unq_curso_posicion UNIQUE (id_curso, id_posicion)
);

CREATE TABLE IF NOT EXISTS materias_bimestre (
    id_materia_bimestre SERIAL PRIMARY KEY,
    id_curso INTEGER NOT NULL REFERENCES cursos(id_curso) ON DELETE CASCADE,
    nombre_materia VARCHAR(255) NOT NULL,
    duracion_bimestres VARCHAR(50) NOT NULL,
    total_horas INTEGER NOT NULL,
    modalidad VARCHAR(255) NOT NULL DEFAULT 'Virtual',
    docente_id INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE RESTRICT,
    lapso_academico SMALLINT NOT NULL DEFAULT 1
);

CREATE TABLE IF NOT EXISTS actividades_config (
    id_actividad_config SERIAL PRIMARY KEY,
    id_materia_bimestre INTEGER NOT NULL REFERENCES materias_bimestre(id_materia_bimestre) ON DELETE CASCADE,
    nombre_actividad VARCHAR(255) NOT NULL,
    ponderacion_porcentaje NUMERIC(5, 2) NOT NULL CHECK (ponderacion_porcentaje > 0 AND ponderacion_porcentaje <= 100)
);

CREATE TABLE IF NOT EXISTS notas_participante (
    id_nota SERIAL PRIMARY KEY,
    id_usuario INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE CASCADE,
    id_actividad_config INTEGER NOT NULL REFERENCES actividades_config(id_actividad_config) ON DELETE CASCADE,
    calificacion_obtenida NUMERIC(4, 2) NOT NULL,
    UNIQUE (id_usuario, id_actividad_config)
);

CREATE TABLE IF NOT EXISTS actas_cierre (
    id_acta SERIAL PRIMARY KEY,
    id_materia_bimestre INTEGER NOT NULL REFERENCES materias_bimestre(id_materia_bimestre) ON DELETE CASCADE,
    fecha_cierre TIMESTAMP WITHOUT TIME ZONE DEFAULT NOW(),
    participantes_inscritos INTEGER NOT NULL DEFAULT 0,
    participantes_aprobados INTEGER NOT NULL DEFAULT 0,
    firmante_vicerrector_id INTEGER REFERENCES usuarios(id) ON DELETE SET NULL,
    firmante_coord_id INTEGER REFERENCES usuarios(id) ON DELETE SET NULL
);

CREATE TABLE IF NOT EXISTS comprobantes_pago (
    id_comprobante SERIAL PRIMARY KEY,
    id_usuario INTEGER NOT NULL REFERENCES usuarios(id) ON DELETE CASCADE,
    id_curso INTEGER NOT NULL REFERENCES cursos(id_curso) ON DELETE CASCADE,
    archivo_ruta VARCHAR(255) NOT NULL,
    numero_operacion VARCHAR(100) NOT NULL,
    banco_origen VARCHAR(100),
    monto NUMERIC(10, 2) NOT NULL,
    estado VARCHAR(50) NOT NULL DEFAULT 'Pendiente',
    fecha_pago DATE NOT NULL,
    fecha_subida TIMESTAMP WITHOUT TIME ZONE DEFAULT NOW()
);

-- ==================== AUDITORÍA (TRIGGERS) ====================

CREATE OR REPLACE FUNCTION cursos_audit() RETURNS trigger
    LANGUAGE plpgsql
    AS $$
DECLARE
    accion_texto VARCHAR(10);
    usuario_sistema INTEGER;
BEGIN
    BEGIN
        SELECT current_setting('myapp.current_user_id', true)::INTEGER INTO usuario_sistema;
    EXCEPTION
        WHEN others THEN
            usuario_sistema := -1; 
    END;

    IF TG_OP = 'DELETE' THEN
        accion_texto := 'DELETE';
        INSERT INTO auditoria (usuario, accion, tabla_afectada, fecha, dato_previo, dato_modificado)
        VALUES (usuario_sistema, accion_texto, TG_TABLE_NAME, now(), OLD::text, NULL);
        RETURN OLD;
    ELSIF TG_OP = 'UPDATE' THEN
        accion_texto := 'UPDATE';
        INSERT INTO auditoria (usuario, accion, tabla_afectada, fecha, dato_previo, dato_modificado)
        VALUES (usuario_sistema, accion_texto, TG_TABLE_NAME, now(), OLD::text, NEW::text);
        RETURN NEW;
    ELSIF TG_OP = 'INSERT' THEN
        accion_texto := 'INSERT';
        INSERT INTO auditoria (usuario, accion, tabla_afectada, fecha, dato_previo, dato_modificado)
        VALUES (usuario_sistema, accion_texto, TG_TABLE_NAME, now(), NULL, NEW::text);
        RETURN NEW;
    END IF;

    RETURN NULL;
END;
$$;

DO $$ 
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'certificaciones_trigger') THEN
        CREATE TRIGGER certificaciones_trigger AFTER INSERT OR DELETE OR UPDATE ON certificaciones FOR EACH ROW EXECUTE PROCEDURE cursos_audit();
    END IF;
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'cursos_trigger') THEN
        CREATE TRIGGER cursos_trigger AFTER INSERT OR DELETE OR UPDATE ON cursos FOR EACH ROW EXECUTE PROCEDURE cursos_audit();
    END IF;
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'roles_trigger') THEN
        CREATE TRIGGER roles_trigger AFTER INSERT OR DELETE OR UPDATE ON roles FOR EACH ROW EXECUTE PROCEDURE cursos_audit();
    END IF;
    IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'usuarios_trigger') THEN
        CREATE TRIGGER usuarios_trigger AFTER INSERT OR DELETE OR UPDATE ON usuarios FOR EACH ROW EXECUTE PROCEDURE cursos_audit();
    END IF;
END $$;

    r RECORD;

    END LOOP;
END $$;
